---
title: AnimatePresence
description: Enter and exit animation examples for conditionally rendered elements.
---

`AnimatePresence` enables exit animations on unmount. Wrap conditional content and add an `exit` prop to each motion child. A `key` prop is required on every direct child.

## Basic fade in/out

The simplest pattern — fade in on mount, fade out on unmount:

```svelte
<script lang="ts">
    import { motion, AnimatePresence } from '@humanspeak/svelte-motion'

    let isVisible = $state(true)
</script>

<AnimatePresence>
    {#if isVisible}
        <motion.div
            key="box"
            initial={{ opacity: 0, scale: 0 }}
            animate={{ opacity: 1, scale: 1 }}
            exit={{ opacity: 0, scale: 0 }}
            transition={{ type: 'spring', stiffness: 300, damping: 25 }}
        >
            Hello
        </motion.div>
    {/if}
</AnimatePresence>

<button onclick={() => (isVisible = !isVisible)}>
    {isVisible ? 'Hide' : 'Show'}
</button>
```

<Note>
  Every motion component that is a direct child of `AnimatePresence` requires a `key` prop. The key is used to track enter vs. exit state and must be a string.
</Note>

## AnimatePresence modes

<Tabs>
  <Tab title="wait">
    `mode="wait"` — the exiting element finishes its exit animation before the entering element starts. Use this for content swaps where overlap looks wrong.

    ```svelte
    <script lang="ts">
        import { AnimatePresence, motion } from '@humanspeak/svelte-motion'

        let selectedTab = $state(0)
        const tabs = ['Home', 'About', 'Contact']
    </script>

    <div class="tabs">
        {#each tabs as tab, i}
            <button onclick={() => (selectedTab = i)}>{tab}</button>
        {/each}
    </div>

    <AnimatePresence mode="wait">
        {#key selectedTab}
            <motion.div
                key="content-{selectedTab}"
                initial={{ y: 10, opacity: 0 }}
                animate={{ y: 0, opacity: 1 }}
                exit={{ y: -10, opacity: 0 }}
                transition={{ duration: 0.2 }}
            >
                {tabs[selectedTab]}
            </motion.div>
        {/key}
    </AnimatePresence>
    ```
  </Tab>
  <Tab title="sync">
    `mode="sync"` (default) — enter and exit animations run at the same time.

    ```svelte
    <script lang="ts">
        import { AnimatePresence, motion } from '@humanspeak/svelte-motion'

        let isVisible = $state(true)
    </script>

    <!-- Default mode — no need to specify mode="sync" explicitly -->
    <AnimatePresence>
        {#if isVisible}
            <motion.div
                key="box"
                initial={{ opacity: 0, x: -20 }}
                animate={{ opacity: 1, x: 0 }}
                exit={{ opacity: 0, x: 20 }}
            />
        {/if}
    </AnimatePresence>
    ```
  </Tab>
  <Tab title="popLayout">
    `mode="popLayout"` — the exiting element is removed from the document flow immediately (absolute positioned), so other elements reflow right away while the exit animation plays.

    ```svelte
    <script lang="ts">
        import { AnimatePresence, motion } from '@humanspeak/svelte-motion'

        let items = $state([1, 2, 3, 4, 5])

        function remove(id: number) {
            items = items.filter(i => i !== id)
        }
    </script>

    <AnimatePresence mode="popLayout">
        {#each items as item (item)}
            <motion.div
                key={String(item)}
                layout
                initial={{ opacity: 0, scale: 0.8 }}
                animate={{ opacity: 1, scale: 1 }}
                exit={{ opacity: 0, scale: 0.8 }}
                onclick={() => remove(item)}
            >
                {item}
            </motion.div>
        {/each}
    </AnimatePresence>
    ```
  </Tab>
</Tabs>

## Multiple children (list)

Pass a keyed `{#each}` block and give each motion component a string `key` matching the item id:

```svelte
<script lang="ts">
    import { motion, AnimatePresence } from '@humanspeak/svelte-motion'

    let items = $state([
        { id: 1, text: 'Item 1' },
        { id: 2, text: 'Item 2' },
        { id: 3, text: 'Item 3' }
    ])

    function removeItem(id: number) {
        items = items.filter(item => item.id !== id)
    }
</script>

<AnimatePresence>
    {#each items as item (item.id)}
        <motion.div
            key={String(item.id)}
            initial={{ opacity: 0, x: -20 }}
            animate={{ opacity: 1, x: 0 }}
            exit={{ opacity: 0, x: 20 }}
            transition={{ type: 'spring', stiffness: 500, damping: 30 }}
        >
            {item.text}
            <button onclick={() => removeItem(item.id)}>Remove</button>
        </motion.div>
    {/each}
</AnimatePresence>
```

## Exit completion callback

Run code after all exit animations finish:

```svelte
<script lang="ts">
    import { AnimatePresence, motion } from '@humanspeak/svelte-motion'

    let isVisible = $state(true)

    function handleExitComplete() {
        console.log('All exits complete')
    }
</script>

<AnimatePresence onExitComplete={handleExitComplete}>
    {#if isVisible}
        <motion.div
            key="box"
            exit={{ opacity: 0 }}
            transition={{ duration: 0.2 }}
        />
    {/if}
</AnimatePresence>
```

## Suppress initial animation

Set `initial={false}` to skip enter animations on the first render:

```svelte
<AnimatePresence initial={false}>
    {#if isVisible}
        <motion.div
            key="box"
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            exit={{ opacity: 0 }}
        />
    {/if}
</AnimatePresence>
```

## With variants

```svelte
<script lang="ts">
    import { motion, AnimatePresence, type Variants } from '@humanspeak/svelte-motion'

    let isVisible = $state(true)

    const itemVariants: Variants = {
        hidden:  { opacity: 0, y: 20 },
        visible: { opacity: 1, y: 0 },
        exit:    { opacity: 0, y: -20 }
    }
</script>

<AnimatePresence>
    {#if isVisible}
        <motion.div
            key="item"
            variants={itemVariants}
            initial="hidden"
            animate="visible"
            exit="exit"
        />
    {/if}
</AnimatePresence>
```

## Modal / dialog pattern

```svelte
<script lang="ts">
    import { AnimatePresence, motion } from '@humanspeak/svelte-motion'

    let showModal = $state(false)
</script>

<AnimatePresence>
    {#if showModal}
        <!-- Backdrop -->
        <motion.div
            key="backdrop"
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            exit={{ opacity: 0 }}
            style="position: fixed; inset: 0; background: rgba(0,0,0,0.5);"
            onclick={() => (showModal = false)}
        />

        <!-- Modal -->
        <motion.div
            key="modal"
            initial={{ opacity: 0, scale: 0.9, y: 20 }}
            animate={{ opacity: 1, scale: 1,   y: 0  }}
            exit={{    opacity: 0, scale: 0.9, y: 20 }}
            transition={{ type: 'spring', stiffness: 300, damping: 25 }}
            style="position: fixed; top: 50%; left: 50%; transform: translate(-50%, -50%);"
        >
            Modal content
        </motion.div>
    {/if}
</AnimatePresence>

<button onclick={() => (showModal = true)}>Open modal</button>
```

## Exit transition precedence

Transition priority from lowest to highest:

1. Base default: `{ duration: 0.35 }`
2. Component `transition` prop
3. `exit.transition` (inline inside the `exit` object)

```svelte
<motion.div
    key="box"
    initial={{ opacity: 0 }}
    animate={{ opacity: 1 }}
    exit={{
        opacity: 0,
        scale: 0.8,
        transition: { duration: 0.2, ease: 'easeIn' }
    }}
/>
```

## Props reference

### `AnimatePresence`

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `mode` | `"sync" \| "wait" \| "popLayout"` | `"sync"` | Controls enter/exit overlap behavior |
| `initial` | `boolean` | `true` | When `false`, children skip their enter animation on first mount |
| `onExitComplete` | `() => void` | — | Called when all exit animations complete |

### Motion component (inside AnimatePresence)

| Prop | Type | Description |
|------|------|-------------|
| `key` | `string` | Required. Unique identifier for tracking enter/exit state |
| `exit` | `object \| string` | Animation state when the component is removed from the DOM |
